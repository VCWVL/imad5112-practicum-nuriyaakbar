1    package com.example.musicplaylstmanager
2    
3    import android.annotation.SuppressLint
4    import android.content.Intent
5    import android.os.Bundle
6    import android.widget.*
7    import androidx.appcompat.app.AppCompatActivity
8    
9    class MainActivity : AppCompatActivity() {
10   
11       companion object {
12           val playlist = mutableListOf<Song>()
13       }
14   
15       @SuppressLint("MissingInflatedId")
16       override fun onCreate(savedInstanceState: Bundle?) {
17           super.onCreate(savedInstanceState)
18           setContentView(R.layout.activity_main)
19   
20           val addButton = findViewById<Button>(R.id.btnAdd)
21           val nextButton = findViewById<Button>(R.id.btnNext)
22           val exitButton = findViewById<Button>(R.id.btnExit)
23   
24           addButton.setOnClickListener {
25               if (playlist.size >= 4) {
26                   Toast.makeText(this, "Only 4 songs allowed.", Toast.LENGTH_SHORT).show()
27                   return@setOnClickListener
28               }
29   
30               val dialogView = layoutInflater.inflate(R.layout.layout, null)
31               val dialogBuilder = android.app.AlertDialog.Builder(this)
32                   .setTitle("Add Song")
33                   .setView(dialogView)
34                   .setPositiveButton("Add") { _, _ ->
35                       val title = dialogView.findViewById<EditText>(R.id.etTitle).text.toString()
36                       val artist = dialogView.findViewById<EditText>(R.id.etArtist).text.toString()
37                       val rating = dialogView.findViewById<EditText>(R.id.etRating).text.toString().toIntOrNull() ?: 0
38                       val comments = dialogView.findViewById<EditText>(R.id.etComments).text.toString()
39   
40                       if (title.isNotBlank() && artist.isNotBlank() && rating in 1..5) {
41                           playlist.add(Song(title, artist, rating, comments))
42                           Toast.makeText(this, "Song added!", Toast.LENGTH_SHORT).show()
43                       } else {
44                           Toast.makeText(this, "Please enter valid details", Toast.LENGTH_SHORT).show()
45                       }
46                   }
47                   .setNegativeButton("Cancel", null)
48   
49               dialogBuilder.show()
50           }
51   
52           nextButton.setOnClickListener {
53               startActivity(Intent(this, DetailActivity::class.java))
54           }
55   
56           exitButton.setOnClickListener {
57               finishAffinity()
58           }
59       }
60   }
61   
62   data class Song(val title: String, val artist: String, val rating: Int, val comments: String)
