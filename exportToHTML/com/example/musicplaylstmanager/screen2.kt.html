1    package com.example.musicplaylstmanager
2    
3    import android.os.Bundle
4    import android.widget.*
5    import androidx.appcompat.app.AppCompatActivity
6    
7    class DetailActivity : AppCompatActivity() {
8    
9        override fun onCreate(savedInstanceState: Bundle?) {
10           super.onCreate(savedInstanceState)
11           setContentView(R.layout.activity_screen2)
12   
13           val showButton = findViewById<Button>(R.id.btnShow)
14           val avgButton = findViewById<Button>(R.id.btnAverage)
15           val returnButton = findViewById<Button>(R.id.btnReturn)
16           val textView = findViewById<TextView>(R.id.tvDetails)
17   
18           showButton.setOnClickListener {
19               val details = StringBuilder()
20               for ((index, song) in MainActivity.playlist.withIndex()) {
21                   details.append("Song ${index + 1}:\nTitle: ${song.title}\nArtist: ${song.artist}\nRating: ${song.rating}\nComments: ${song.comments}\n\n")
22               }
23               textView.text = details.toString()
24           }
25   
26           avgButton.setOnClickListener {
27               val totalRating = MainActivity.playlist.sumOf { it.rating }
28               val avg = if (MainActivity.playlist.isNotEmpty()) totalRating.toDouble() / MainActivity.playlist.size else 0.0
29               textView.text = "Average Rating: %.2f".format(avg)
30           }
31   
32           returnButton.setOnClickListener {
33               finish()
34           }
35       }
36   }
